## 1. ¿Por qué TypeScript?

### Ventajas

| Característica | JavaScript | TypeScript |
|----------------|------------|------------|
| Tipado | Dinámico | Estático |
| Errores | En runtime | En compilación |
| Autocompletado | Limitado | Excelente |
| Refactoring | Riesgoso | Seguro |
| Documentación | Manual | Integrada en tipos |
| Curva aprendizaje | Baja | Media |

### Ejemplo: Errores detectados antes

```javascript
// JavaScript - Error en runtime
function calculateTotal(items) {
  return items.reduce((sum, item) => sum + item.price, 0);
}

calculateTotal("not an array"); // Runtime Error!
```

```typescript
// TypeScript - Error en compilación
interface Item {
  name: string;
  price: number;
}

function calculateTotal(items: Item[]): number {
  return items.reduce((sum, item) => sum + item.price, 0);
}

calculateTotal("not an array");
// ❌ Error: Argument of type 'string' is not assignable to parameter of type 'Item[]'
```


```json
{
  "compilerOptions": {
```

## 3. Tipos Básicos

### Tipos primitivos

```typescript
// Strings
const name: string = "Ana";

// Numbers
const age: number = 25;
const price: number = 19.99;

// Booleans
const isActive: boolean = true;

// Null y Undefined
const nothing: null = null;
const notDefined: undefined = undefined;

// Any (evitar cuando sea posible)
let flexible: any = "puede ser cualquier cosa";
flexible = 42;

// Unknown (más seguro que any)
let userInput: unknown = "algo";
if (typeof userInput === "string") {
  console.log(userInput.toUpperCase()); // OK, TypeScript sabe que es string
}
```

### Arrays y Tuplas

```typescript
// Arrays
const numbers: number[] = [1, 2, 3];
const names: Array<string> = ["Ana", "Carlos"];

// Arrays readonly
const readonlyArray: readonly number[] = [1, 2, 3];
// readonlyArray.push(4); // ❌ Error

// Tuplas (arrays con tipos fijos por posición)
const coordinate: [number, number] = [40.4168, -3.7038];
const userRecord: [number, string, boolean] = [1, "Ana", true];

// Tupla con labels (más legible)
type Point = [x: number, y: number];
const point: Point = [10, 20];
```

### Objetos

```typescript
// Objeto con tipo inline
const user: { name: string; age: number } = {
  name: "Ana",
  age: 25
};

// Propiedades opcionales
const config: { host: string; port?: number } = {
  host: "localhost"
  // port es opcional
};

// Readonly
const settings: { readonly apiKey: string } = {
  apiKey: "abc123"
};
// settings.apiKey = "new"; // ❌ Error
```

### Uniones y Literales

```typescript
// Union types
let id: string | number;
id = "abc";
id = 123;

// Literal types
type Status = "pending" | "active" | "completed";
const orderStatus: Status = "active";
// orderStatus = "invalid"; // ❌ Error

// Union con null
function findUser(id: number): User | null {
  // ...
}

// Discriminated unions
type Result<T> =
  | { success: true; data: T }
  | { success: false; error: string };

function handleResult(result: Result<User>) {
  if (result.success) {
    console.log(result.data); // TypeScript sabe que es User
  } else {
    console.log(result.error); // TypeScript sabe que es string
  }
}
```

## 4. Interfaces y Types

### Interfaces

```typescript
// Definir interface
interface User {
  id: number;
  name: string;
  email: string;
  age?: number;              // Opcional
  readonly createdAt: Date;  // Solo lectura
}

// Usar interface
const user: User = {
  id: 1,
  name: "Ana",
  email: "ana@example.com",
  createdAt: new Date()
};

// Extender interfaces
interface Admin extends User {
  permissions: string[];
}

// Múltiple herencia
interface Employee extends User, Timestamps {
  department: string;
}
```

### Type Aliases

```typescript
// Type alias básico
type ID = string | number;

// Type alias para objetos
type Point = {
  x: number;
  y: number;
};

// Intersection types (combinar tipos)
type Timestamps = {
  createdAt: Date;
  updatedAt: Date;
};

type UserWithTimestamps = User & Timestamps;

// Tipos utilitarios
type PartialUser = Partial<User>;        // Todos opcionales
type RequiredUser = Required<User>;       // Todos requeridos
type UserName = Pick<User, "name" | "email">;
type UserWithoutId = Omit<User, "id">;
type ReadonlyUser = Readonly<User>;
```

### Interface vs Type

```typescript
// Interface: puede extenderse y fusionarse
interface User {
  name: string;
}
interface User {
  age: number;
}
// User ahora tiene name y age (declaration merging)

// Type: más flexible para unions y tipos complejos
type Response = Success | Error;
type Callback = (data: string) => void;

// Recomendación:
// - Interface para objetos y clases
// - Type para uniones, intersecciones, tipos complejos
```

### Funciones tipadas

```typescript
// Función básica
function greet(name: string): string {
  return `Hello, ${name}!`;
}

// Arrow function
const add = (a: number, b: number): number => a + b;

// Parámetros opcionales y defaults
function createUser(
  name: string,
  age?: number,

  // ...

// Rest parameters

  return numbers.reduce((a, b) => a + b, 0);

}
```

## 5. Genéricos

### Concepto básico

```typescript
// Sin genéricos: perdemos información de tipos
function identity(arg: any): any {
  return arg;
}

// Con genéricos: preservamos el tipo
function identity<T>(arg: T): T {
  return arg;
}

const num = identity<number>(42);    // num es number
const str = identity("hello");        // str es string (inferido)
```

### Genéricos en interfaces

```typescript
// Respuesta API genérica
interface ApiResponse<T> {
  data: T;
  status: number;
  message: string;
}

// Uso
type UserResponse = ApiResponse<User>;
type UsersResponse = ApiResponse<User[]>;

// Con múltiples tipos
interface KeyValuePair<K, V> {
  key: K;
  value: V;
}

const pair: KeyValuePair<string, number> = {
  key: "age",
  value: 25
};
```

### Constraints (restricciones)

```typescript
// Restringir a tipos con cierta propiedad
interface HasId {
  id: number;
}

function findById<T extends HasId>(items: T[], id: number): T | undefined {
  return items.find(item => item.id === id);
}

// Restringir a claves de un objeto
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
  return obj[key];
}

const user = { name: "Ana", age: 25 };
const name = getProperty(user, "name"); // string
// getProperty(user, "invalid"); // ❌ Error
```

### Genéricos en clases

```typescript
class Repository<T extends HasId> {
  private items: T[] = [];

  add(item: T): void {
    this.items.push(item);
  }

  findById(id: number): T | undefined {
    return this.items.find(item => item.id === id);
  }

  getAll(): T[] {
    return [...this.items];
  }
}

const userRepo = new Repository<User>();
userRepo.add({ id: 1, name: "Ana", email: "ana@example.com" });
```

### Middleware tipado

```typescript
// src/middleware/auth.middleware.ts
import { Request, Response, NextFunction } from 'express';
import { verifyToken } from '../utils/jwt';
import { User } from '../models/user';

export const authMiddleware = async (
  req: Request,
  res: Response,
  next: NextFunction
): Promise<void> => {
  try {
    const token = req.headers.authorization?.split(' ')[1];

    if (!token) {
      res.status(401).json({ error: 'Token requerido' });
      return;
    }

    const decoded = verifyToken(token);
    const user = await User.findById(decoded.id);

    if (!user) {
      res.status(401).json({ error: 'Usuario no encontrado' });
      return;
    }

    req.user = user;
    next();
  } catch (error) {
    res.status(401).json({ error: 'Token inválido' });
  }
};
```

### Error Handler tipado

```typescript
// src/middleware/error.middleware.ts
import { Request, Response, NextFunction } from 'express';

interface AppError extends Error {
  status?: number;
  code?: string;
}

export const errorHandler = (
  err: AppError,
  req: Request,
  res: Response,
  next: NextFunction
): void => {
  console.error(err);

  const status = err.status || 500;
  const message = err.message || 'Error interno';

  res.status(status).json({
    error: true,
    message,
    ...(process.env.NODE_ENV === 'development' && { stack: err.stack })
  });
};
```

```typescript
    email: { type: String, required: true, unique: true },
    password: { type: String, required: true },
    role: { type: String, enum: ['user', 'admin'], default: 'user' }
  },
  { timestamps: true }
);

// Método de instancia
userSchema.methods.comparePassword = async function(
  candidatePassword: string
): Promise<boolean> {
  // Implementación...
  return true;
};

// Método estático
userSchema.statics.findByEmail = function(
  email: string

  return this.findOne({ email });

export const User = mongoose.model<IUserDocument, IUserModel>('User', userSchema);
```

## 8. Prisma con TypeScript

### Prisma genera tipos automáticamente

```typescript
// Los tipos se generan desde schema.prisma
// Después de: npx prisma generate

import { PrismaClient, User, Post } from '@prisma/client';

const prisma = new PrismaClient();

async function getUsers(): Promise<User[]> {
  return prisma.user.findMany();

async function createUser(data: { name: string; email: string }): Promise<User> {
  return prisma.user.create({ data });

async function getUserWithPosts(id: number) {
  return prisma.user.findUnique({
    where: { id },
    include: { posts: true }
  });
}
// Retorna: User & { posts: Post[] }
```

## 9. Patrones Comunes

### Result Type (manejo de errores)

```typescript
// Tipo para resultados que pueden fallar
type Result<T, E = Error> =
  | { success: true; data: T }
  | { success: false; error: E };

// Uso
async function findUser(id: number): Promise<Result<User>> {
  try {
    const user = await prisma.user.findUnique({ where: { id } });
    if (!user) {
      return { success: false, error: new Error('Usuario no encontrado') };
    }
    return { success: true, data: user };
  } catch (error) {
    return { success: false, error: error as Error };
  }
}

// Manejo
const result = await findUser(1);
if (result.success) {
  console.log(result.data.name); // TypeScript sabe que es User
} else {
  console.error(result.error.message);
}
```

### DTO Pattern

```typescript
// Data Transfer Objects
interface CreateUserDTO {
  name: string;
  email: string;
  password: string;
}

interface UpdateUserDTO {
  name?: string;
  email?: string;
}

interface UserResponseDTO {
  id: number;
  name: string;
  email: string;
  createdAt: Date;
}

// Función para transformar
function toUserResponse(user: User): UserResponseDTO {
  return {
    id: user.id,
    name: user.name,
    email: user.email,
    createdAt: user.createdAt
  };
}
```

### Repository Pattern

```typescript
// Interface genérica
interface IRepository<T, CreateDTO, UpdateDTO> {
  findAll(): Promise<T[]>;
  findById(id: number): Promise<T | null>;
  create(data: CreateDTO): Promise<T>;
  update(id: number, data: UpdateDTO): Promise<T>;
  delete(id: number): Promise<void>;
}

// Implementación
class UserRepository implements IRepository<User, CreateUserDTO, UpdateUserDTO> {
  async findAll(): Promise<User[]> {
    return prisma.user.findMany();
  }

  async findById(id: number): Promise<User | null> {
    return prisma.user.findUnique({ where: { id } });
  }

  async create(data: CreateUserDTO): Promise<User> {
    return prisma.user.create({ data });
  }

  async update(id: number, data: UpdateUserDTO): Promise<User> {
    return prisma.user.update({ where: { id }, data });
  }

  async delete(id: number): Promise<void> {
    await prisma.user.delete({ where: { id } });
  }
}
```

### Type Guards

```typescript
// Type guard con is
function isUser(obj: unknown): obj is User {
  return (
    typeof obj === 'object' &&
    obj !== null &&
    'id' in obj &&
    'name' in obj &&
    'email' in obj
  );
}

// Uso
function processData(data: unknown) {
  if (isUser(data)) {
    console.log(data.name); // TypeScript sabe que es User
  }
}

// Type guard para errores
function isError(error: unknown): error is Error {
  return error instanceof Error;
}
```

## 10. Migrar de JavaScript a TypeScript

### Estrategia gradual

1. **Instalar TypeScript** sin cambiar código
   ```bash
   npm install -D typescript @types/node
   ```

2. **Configurar tsconfig.json** permisivo
   ```json
   {
     "compilerOptions": {
       "allowJs": true,
       "checkJs": false,
       "strict": false
     }
   }
   ```

3. **Renombrar archivos** uno a uno (.js → .ts)

4. **Añadir tipos** gradualmente

5. **Activar strict mode** cuando todo esté tipado

### Checklist de migración

```markdown
- [ ] Instalar TypeScript y tipos
- [ ] Crear tsconfig.json
- [ ] Renombrar index.js → index.ts
- [ ] Añadir tipos a las funciones principales
```

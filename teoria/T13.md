## 1. Introducción a la Gestión de Archivos

### El Desafío del Manejo de Archivos
Las aplicaciones web modernas necesitan gestionar archivos: imágenes de perfil, documentos, videos, etc. Esto presenta varios desafíos:

```
┌─────────────────────────────────────────────────────────────┐
│                  Flujo de Subida de Archivos                │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│   Cliente              Servidor              Storage        │
│      │                    │                     │           │
│      │  multipart/form    │                     │           │
│      │ ──────────────────>│                     │           │
│      │                    │                     │           │
│      │                    │ ── Validar ──>      │           │
│      │                    │ ── Procesar ──>     │           │
│      │                    │ ── Optimizar ──>    │           │
│      │                    │                     │           │
│      │                    │     Upload          │           │
│      │                    │ ──────────────────> │           │
│      │                    │                     │           │
│      │                    │ <── URL pública ──  │           │
│      │  URL del archivo   │                     │           │
│      │ <──────────────────│                     │           │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### Opciones de Almacenamiento

| Opción | Pros | Contras |
|--------|------|---------|
| **Local (filesystem)** | Simple, sin costos extra | No escala, se pierde en deploys |
| **Cloudinary** | CDN, transformaciones automáticas, fácil | Costos por transformación |
| **Cloudflare R2** | Sin costos de egreso, API compatible con S3, económico | Sin transformaciones de imagen nativas |
| **AWS S3** | Escalable, estándar de la industria | Costos de egreso, más configuración |
| **Supabase Storage** | Integrado con Supabase | Dependencia del ecosistema |

```javascript
// src/config/multer.js
import multer from 'multer';
import path from 'path';
import crypto from 'crypto';

// Almacenamiento en disco
const diskStorage = multer.diskStorage({
  destination: (req, file, cb) => {
    cb(null, 'uploads/');
  },
  filename: (req, file, cb) => {
    // Generar nombre único
    const uniqueSuffix = crypto.randomBytes(16).toString('hex');
    const ext = path.extname(file.originalname);
    cb(null, `${uniqueSuffix}${ext}`);
  }
});

// Almacenamiento en memoria (para subir a cloud)
const memoryStorage = multer.memoryStorage();

export const uploadDisk = multer({ storage: diskStorage });
export const uploadMemory = multer({ storage: memoryStorage });
```

### Validación de Archivos

```javascript
// src/config/multer.js
import multer from 'multer';

// Tipos MIME permitidos
const ALLOWED_MIMES = {
  image: ['image/jpeg', 'image/png', 'image/gif', 'image/webp'],
  document: ['application/pdf', 'application/msword',
             'application/vnd.openxmlformats-officedocument.wordprocessingml.document'],
  video: ['video/mp4', 'video/webm']
};

// Tamaños máximos por tipo (en bytes)
const MAX_SIZES = {
  image: 5 * 1024 * 1024,      // 5 MB
  document: 10 * 1024 * 1024,  // 10 MB
  video: 100 * 1024 * 1024     // 100 MB
};

// Filtro de archivos
const createFileFilter = (type) => (req, file, cb) => {
  if (ALLOWED_MIMES[type]?.includes(file.mimetype)) {
    cb(null, true);
  } else {
    cb(new Error(`Tipo de archivo no permitido. Solo: ${ALLOWED_MIMES[type].join(', ')}`), false);
  }
};

// Crear configuración de multer para imágenes
export const uploadImage = multer({
  storage: multer.memoryStorage(),
  limits: {
    fileSize: MAX_SIZES.image,
    files: 5 // Máximo 5 archivos
  },
  fileFilter: createFileFilter('image')
});

// Para documentos
export const uploadDocument = multer({
  storage: multer.memoryStorage(),
  limits: { fileSize: MAX_SIZES.document },
  fileFilter: createFileFilter('document')
});

// Para videos
export const uploadVideo = multer({
  storage: multer.memoryStorage(),
  limits: { fileSize: MAX_SIZES.video },
  fileFilter: createFileFilter('video')
});
```

```javascript
// src/middleware/uploadError.js
import multer from 'multer';

export const handleUploadError = (err, req, res, next) => {
  if (err instanceof multer.MulterError) {
    // Errores de Multer
    switch (err.code) {
      case 'LIMIT_FILE_SIZE':
        return res.status(400).json({
          error: 'Archivo demasiado grande',
          maxSize: '5 MB'
        });
      case 'LIMIT_FILE_COUNT':
        return res.status(400).json({
          error: 'Demasiados archivos',
          maxFiles: 10
        });
      case 'LIMIT_UNEXPECTED_FILE':
        return res.status(400).json({
          error: `Campo no esperado: ${err.field}`
        });
      default:
        return res.status(400).json({ error: err.message });
    }
  }

  if (err.message?.includes('Tipo de archivo')) {
    return res.status(400).json({ error: err.message });
  }

  next(err);
};

// En app.js
app.use(handleUploadError);
```

```javascript
// src/services/cloudinary.service.js
import cloudinary from '../config/cloudinary.js';
import { Readable } from 'stream';

class CloudinaryService {
  /**
   * Subir archivo desde buffer (de Multer memoryStorage)
   */
  async uploadBuffer(buffer, options = {}) {
    return new Promise((resolve, reject) => {
      const uploadStream = cloudinary.uploader.upload_stream(
        {
          folder: options.folder || 'uploads',
          resource_type: options.resourceType || 'auto',
          public_id: options.publicId,
          transformation: options.transformation,
          ...options
        },
        (error, result) => {
          if (error) reject(error);
          else resolve(result);
        }
      );

      // Convertir buffer a stream y enviarlo
      const readableStream = Readable.from(buffer);
      readableStream.pipe(uploadStream);
    });
  }

  /**
   * Subir imagen con transformaciones automáticas
   */
  async uploadImage(buffer, options = {}) {
    return this.uploadBuffer(buffer, {
      folder: 'images',
      resourceType: 'image',
      transformation: [
        { quality: 'auto:good' },
        { fetch_format: 'auto' }
      ],
      ...options
    });
  }

  /**
   * Subir avatar con recorte circular
   */
  async uploadAvatar(buffer, userId) {
    return this.uploadBuffer(buffer, {
      folder: 'avatars',
      public_id: `user_${userId}`,
      overwrite: true,
      transformation: [
        { width: 300, height: 300, crop: 'fill', gravity: 'face' },
        { radius: 'max' },
        { quality: 'auto' }
      ]
    });
  }

  /**
   * Subir imagen de producto con múltiples tamaños
   */
  async uploadProductImage(buffer, productId, index = 0) {
    return this.uploadBuffer(buffer, {
      folder: 'products',
      public_id: `product_${productId}_${index}`,
      eager: [
        { width: 150, height: 150, crop: 'fill' },   // Thumbnail
        { width: 400, height: 400, crop: 'fill' },   // Card
        { width: 800, height: 800, crop: 'limit' }   // Detail
      ],
      eager_async: true
    });
  }

  /**
   * Eliminar archivo por public_id
   */
  async delete(publicId, resourceType = 'image') {
    return cloudinary.uploader.destroy(publicId, {
      resource_type: resourceType
    });
  }

  /**
   * Eliminar múltiples archivos
   */
  async deleteMany(publicIds, resourceType = 'image') {
    return cloudinary.api.delete_resources(publicIds, {
      resource_type: resourceType
    });
  }

  /**
   * Obtener URL optimizada
   */
  getOptimizedUrl(publicId, options = {}) {
    return cloudinary.url(publicId, {
      fetch_format: 'auto',
      quality: 'auto',
      ...options
    });
  }

  /**
   * Generar URL con transformaciones
   */
  getTransformedUrl(publicId, transformations) {
    return cloudinary.url(publicId, {
      transformation: transformations
    });
  }
}

export default new CloudinaryService();
```

```javascript
export const uploadProductImages = async (req, res, next) => {
  try {
    const { productId } = req.params;

    if (!req.files?.length) {
      return res.status(400).json({ error: 'No se enviaron imágenes' });
    }

    // Subir todas las imágenes en paralelo
    const uploadPromises = req.files.map((file, index) =>
      cloudinaryService.uploadProductImage(
        file.buffer,
        productId,
        index
      )
    );

    const results = await Promise.all(uploadPromises);

    // Extraer URLs
    const images = results.map(r => ({
      url: r.secure_url,
      publicId: r.public_id,
      thumbnails: r.eager?.map(e => e.secure_url) || []
    }));

    res.json({
      message: `${images.length} imágenes subidas`,
      images
    });
  } catch (error) {
    next(error);
  }
};

export const deleteImage = async (req, res, next) => {
```

## 4. Almacenamiento Compatible con S3 (AWS S3 / Cloudflare R2)

Tanto **AWS S3** como **Cloudflare R2** usan la misma API (S3-compatible), así que el código es prácticamente idéntico. La principal diferencia es la configuración del endpoint.

| Característica | AWS S3 | Cloudflare R2 |
|---------------|--------|---------------|
| **Costos de egreso** | Sí (puede ser significativo) | Gratis |
| **Precio almacenamiento** | ~$0,023/GB/mes | ~$0,015/GB/mes |
| **API** | S3 nativa | Compatible con S3 |
| **Transformaciones** | No (necesitas Lambda) | No (necesitas Workers) |
| **CDN integrado** | CloudFront (separado) | Incluido |
| **Tier gratuito** | 5 GB, 12 meses | 10 GB, sin límite de tiempo |

### Instalación

```bash

```

### Servicio de Storage (S3/R2)

```javascript
// src/services/s3.service.js — Compatible con AWS S3 y Cloudflare R2
import {
  PutObjectCommand,
  GetObjectCommand,
  DeleteObjectCommand,
  ListObjectsV2Command
} from '@aws-sdk/client-s3';
import { getSignedUrl } from '@aws-sdk/s3-request-presigner';
import { s3Client, BUCKET_NAME } from '../config/s3.js';
import crypto from 'crypto';
import path from 'path';

class S3Service {
  /**
   * Subir archivo a S3
   */
  async upload(buffer, options = {}) {
    const {
      folder = 'uploads',
      filename,
      contentType,
      isPublic = true
    } = options;

    // Generar nombre único si no se proporciona
    const key = filename
      ? `${folder}/${filename}`
      : `${folder}/${crypto.randomBytes(16).toString('hex')}${options.extension || ''}`;

    const command = new PutObjectCommand({
      Bucket: BUCKET_NAME,
      Key: key,
      Body: buffer,
      ContentType: contentType,
      ACL: isPublic ? 'public-read' : 'private'
    });

    await s3Client.send(command);

    return {
      key,
      url: isPublic
        ? `https://${BUCKET_NAME}.s3.amazonaws.com/${key}`
        : null,
      bucket: BUCKET_NAME
    };
  }

  /**
   * Subir imagen
   */
  async uploadImage(buffer, options = {}) {
    return this.upload(buffer, {
      folder: 'images',
      extension: '.jpg',
      contentType: options.contentType || 'image/jpeg',
      ...options
    });
  }

  /**
   * Obtener URL firmada (para archivos privados)
   */
  async getSignedUrl(key, expiresIn = 3600) {
    const command = new GetObjectCommand({
      Bucket: BUCKET_NAME,
      Key: key
    });

    return getSignedUrl(s3Client, command, { expiresIn });
  }

  /**
   * Generar URL para subida directa desde cliente
   */
  async getUploadUrl(key, contentType, expiresIn = 300) {
    const command = new PutObjectCommand({
      Bucket: BUCKET_NAME,
      Key: key,
      ContentType: contentType
    });

    return getSignedUrl(s3Client, command, { expiresIn });
  }

  /**
   * Eliminar archivo
   */
  async delete(key) {
    const command = new DeleteObjectCommand({
      Bucket: BUCKET_NAME,
      Key: key
    });

    await s3Client.send(command);
    return { deleted: key };
  }

  /**
   * Listar archivos en una carpeta
   */
  async listFiles(prefix, maxKeys = 100) {
    const command = new ListObjectsV2Command({
      Bucket: BUCKET_NAME,
      Prefix: prefix,
      MaxKeys: maxKeys
    });

    const response = await s3Client.send(command);

    return response.Contents?.map(item => ({
      key: item.Key,
      size: item.Size,
      lastModified: item.LastModified,
      url: `https://${BUCKET_NAME}.s3.amazonaws.com/${item.Key}`
    })) || [];
  }
}

export default new S3Service();
```

### Sharp: Optimización de Imágenes

```javascript
// src/services/image.service.js
import sharp from 'sharp';

class ImageService {
  /**
   * Redimensionar imagen
   */
  async resize(buffer, width, height, options = {}) {
    return sharp(buffer)
      .resize(width, height, {
        fit: options.fit || 'cover',
        position: options.position || 'center'
      })
      .toBuffer();
  }

  /**
   * Optimizar imagen para web
   */
  async optimize(buffer, options = {}) {
    const {
      format = 'webp',
      quality = 80,
      maxWidth = 1920,
      maxHeight = 1080
    } = options;

    let pipeline = sharp(buffer)
      .resize(maxWidth, maxHeight, {
        fit: 'inside',
        withoutEnlargement: true
      });

    // Aplicar formato
    switch (format) {
      case 'webp':
        pipeline = pipeline.webp({ quality });
        break;
      case 'jpeg':
        pipeline = pipeline.jpeg({ quality, mozjpeg: true });
        break;
      case 'png':
        pipeline = pipeline.png({ compressionLevel: 9 });
        break;
      case 'avif':
        pipeline = pipeline.avif({ quality });
        break;
    }

    return pipeline.toBuffer();
  }

  /**
   * Generar thumbnail
   */
  async thumbnail(buffer, size = 150) {
    return sharp(buffer)
      .resize(size, size, { fit: 'cover' })
      .webp({ quality: 70 })
      .toBuffer();
  }

  /**
   * Generar múltiples tamaños
   */
  async generateSizes(buffer, sizes = []) {
    const defaultSizes = [
      { name: 'thumbnail', width: 150, height: 150 },
      { name: 'small', width: 400, height: 400 },
      { name: 'medium', width: 800, height: 800 },
      { name: 'large', width: 1200, height: 1200 }
    ];

    const sizeConfigs = sizes.length ? sizes : defaultSizes;

    const results = await Promise.all(
      sizeConfigs.map(async (size) => {
        const resized = await sharp(buffer)
          .resize(size.width, size.height, {
            fit: 'inside',
            withoutEnlargement: true
          })
          .webp({ quality: 80 })
          .toBuffer();

        return {
          name: size.name,
          buffer: resized,
          width: size.width,
          height: size.height
        };
      })
    );

    return results;
  }

  /**
   * Obtener metadatos de imagen
   */
  async getMetadata(buffer) {
    return sharp(buffer).metadata();
  }

  /**
   * Convertir a formato WebP
   */
  async toWebP(buffer, quality = 80) {
    return sharp(buffer)
      .webp({ quality })
      .toBuffer();
  }

  /**
   * Rotar imagen
   */
  async rotate(buffer, angle) {
    return sharp(buffer)
      .rotate(angle)
      .toBuffer();
  }

  /**
   * Añadir marca de agua
   */
  async addWatermark(imageBuffer, watermarkBuffer, options = {}) {
    const {
      gravity = 'southeast',
      opacity = 0.5
    } = options;

    // Obtener dimensiones de la imagen
    const metadata = await sharp(imageBuffer).metadata();

    // Redimensionar marca de agua
    const watermark = await sharp(watermarkBuffer)
      .resize(Math.round(metadata.width * 0.2))
      .ensureAlpha(opacity)
      .toBuffer();

    return sharp(imageBuffer)
      .composite([{
        input: watermark,
        gravity
      }])
      .toBuffer();
  }
}

export default new ImageService();
```

```javascript
// Verificar tipo real del archivo (no solo la extensión)
export const validateFileType = (allowedTypes) => async (req, res, next) => {
  try {
    if (!req.file) return next();

    // Detectar tipo real del archivo por su contenido
    const type = await fileType.fromBuffer(req.file.buffer);

    if (!type || !allowedTypes.includes(type.mime)) {
      return res.status(400).json({
        error: 'Tipo de archivo no válido',
        detected: type?.mime || 'desconocido',
        allowed: allowedTypes
      });
    }

    // Adjuntar tipo real al request
    req.file.detectedType = type;
    next();
  } catch (error) {
    next(error);
  }
};

// Verificar dimensiones de imagen
export const validateImageDimensions = (maxWidth, maxHeight) => async (req, res, next) => {
  try {
    if (!req.file) return next();

    const sharp = (await import('sharp')).default;
    const metadata = await sharp(req.file.buffer).metadata();

    if (metadata.width > maxWidth || metadata.height > maxHeight) {
      return res.status(400).json({
        error: `Imagen muy grande. Máximo: ${maxWidth}x${maxHeight}px`,
        actual: `${metadata.width}x${metadata.height}px`
      });
    }

    req.file.metadata = metadata;
    next();
  } catch (error) {
    next(error);
  }
};

// Escanear contenido malicioso (básico)
export const scanForMalware = async (req, res, next) => {
  try {
    if (!req.file) return next();

    const content = req.file.buffer.toString('utf8', 0, 1000);

    // Patrones sospechosos
    const suspiciousPatterns = [
      /<script/i,
      /javascript:/i,
      /on\w+\s*=/i,  // onclick, onerror, etc.
      /<\?php/i,
      /<%.*%>/
    ];

    for (const pattern of suspiciousPatterns) {
      if (pattern.test(content)) {
        return res.status(400).json({
          error: 'Archivo rechazado por contenido sospechoso'
        });
      }
    }

    next();
  } catch (error) {
    next(error);
  }
};

// Limitar uploads por usuario
const uploadCounts = new Map();

export const rateLimit = (maxUploads, windowMs) => (req, res, next) => {
  const userId = req.user?._id?.toString() || req.ip;
  const now = Date.now();

  if (!uploadCounts.has(userId)) {
    uploadCounts.set(userId, []);
  }

  const userUploads = uploadCounts.get(userId);

  // Limpiar uploads antiguos
  const recentUploads = userUploads.filter(time => now - time < windowMs);
  uploadCounts.set(userId, recentUploads);

  if (recentUploads.length >= maxUploads) {
    return res.status(429).json({
      error: 'Demasiadas subidas. Intenta más tarde.',
      retryAfter: Math.ceil((recentUploads[0] + windowMs - now) / 1000)
    });
  }

  recentUploads.push(now);
  next();
};

// Generar hash para deduplicación
export const generateFileHash = async (req, res, next) => {
  try {
    if (!req.file) return next();

    const hash = createHash('sha256')
      .update(req.file.buffer)
      .digest('hex');

    req.file.hash = hash;
    next();
  } catch (error) {
    next(error);
  }
};
```

```javascript
// Comprimir imagen antes de enviar
async function compressImage(file, maxWidth = 1920, quality = 0.8) {
  return new Promise((resolve) => {
    const reader = new FileReader();

    reader.onload = (e) => {
      const img = new Image();

      img.onload = () => {
        const canvas = document.createElement('canvas');

        let { width, height } = img;

        // Redimensionar si es muy grande
        if (width > maxWidth) {
          height = (height * maxWidth) / width;
          width = maxWidth;
        }

        canvas.width = width;
        canvas.height = height;

        const ctx = canvas.getContext('2d');
        ctx.drawImage(img, 0, 0, width, height);

        canvas.toBlob(
          (blob) => {
            resolve(new File([blob], file.name, {
              type: 'image/jpeg',
              lastModified: Date.now()
            }));
          },
          'image/jpeg',
          quality
        );
      };

      img.src = e.target.result;
    };

    reader.readAsDataURL(file);
  });
}

// Uso
async function handleFiles(files) {
  const compressed = await Promise.all(
    Array.from(files).map(file => compressImage(file))
  );

  // Ahora subir los archivos comprimidos
  uploadFiles(compressed);
}
```
